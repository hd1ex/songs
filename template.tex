% This template creates songbooks using the songs package
% See http://songs.sourceforge.net/songsdoc/songs.html

\documentclass[landscape, 11pt, a4paper, oneside, ngerman]{book}
\usepackage{geometry}
\usepackage[bookmarks]{hyperref}
\usepackage[chorded]{songs}
\usepackage[utf8x]{inputenc}
\usepackage{textcomp} % Text Companion fonts allows symbols
\usepackage{lastpage} % Page numbers
\usepackage{lmodern} % Latin Modern font
\usepackage{makeidx}

\hypersetup{
    pdftitle={An open-source LaTeX songbook}
}

\title{
    An open-source LaTeX songbook
    \\\vspace{1em}\large
    Contribute at \texttt{\href{https://github.com/hd1ex/songs}{https://github.com/hd1ex/songs}}
}
\author{Made by Alex and friends \ensuremath\heartsuit}

% Use A4 in Landscape and don't waste that much paper space
\geometry{a4paper, landscape, top=15mm, left=15mm, right=15mm, bottom=15mm, headsep=2.5mm, footskip=5mm}

% Disable page numbering
\pagenumbering{gobble}

% Use two columns for song pages
\songcolumns{2}

% Use other fonts
\renewcommand{\familydefault}{\sfdefault}
\renewcommand{\lyricfont}{\sffamily}
\renewcommand{\chorusfont}{\sffamily}
\renewcommand{\idxtitlefont}{\sffamily\normalfont}

\newindex{titleidx}{cbtitle}
\indexsongsas{titleidx}{\thepage}

% Always let songs be on their own pages
\spenalty=-10000

% Use this to display no song numbers
%\nosongnumbers

\newindex{mainindex}{idxfile}

\begin{document}
    \maketitle

    \showindex[2]{Title index}{mainindex}

    \begin{songs}{mainindex}
